document.addEventListener("DOMContentLoaded", function () { const registerForm = document.getElementById("register-form"); if (!registerForm) return; registerForm.addEventListener("submit", async function (event) { event.preventDefault(); const username = document.getElementById("username").value.trim(); const email = document.getElementById("email").value.trim(); const password = document.getElementById("password").value; const preferredAssistant = document.getElementById("preferred_assistant").value; try { // Create the user using Firebase Authentication const userCredential = await firebase.auth().createUserWithEmailAndPassword(email, password); const user = userCredential.user; if (!user) throw new Error("Registration failed. No user object returned."); // Get Firebase ID token for authentication with Flask const idToken = await user.getIdToken(); // Send user data to the Flask backend const response = await fetch("/register", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ idToken: idToken, username: username, preferred_assistant: preferredAssistant // Optionally include email if Flask route supports it: // email: email }) }); const result = await response.json(); if (!response.ok) { throw new Error(result.error || "Registration failed."); } alert("Registration successful! Redirecting to onboarding..."); window.location.href = "/onboarding"; } catch (error) { console.error("Registration error:", error); alert("Error: " + error.message); } }); });
