document.addEventListener("DOMContentLoaded", function () { const registerForm = document.getElementById("register-form"); if (!registerForm) { console.error("Register form not found."); return; } registerForm.addEventListener("submit", async function (event) { event.preventDefault(); const username = document.getElementById("username")?.value.trim(); const email = document.getElementById("email")?.value.trim(); const password = document.getElementById("password")?.value; const preferredAssistant = document.getElementById("preferred_assistant")?.value; if (!username || !email || !password || !preferredAssistant) { alert("Please fill in all required fields."); return; } try { // Firebase registration const userCredential = await firebase.auth().createUserWithEmailAndPassword(email, password); const user = userCredential.user; if (!user) throw new Error("User registration failed."); const idToken = await user.getIdToken(); // Send data to Flask backend const response = await fetch("/register", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ idToken: idToken, username: username, preferred_assistant: preferredAssistant, email: email // Optional: include only if Flask uses it }) }); const result = await response.json(); if (!response.ok) { const errorMessage = result.error || "Registration failed."; throw new Error(errorMessage); } alert("Registration successful! Redirecting..."); window.location.href = "/onboarding"; // Update if using a different route } catch (error) { console.error("Error during registration:", error); alert("Error: " + error.message); } }); });
